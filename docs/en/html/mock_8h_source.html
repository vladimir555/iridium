514auto Mock<TClass>::call(TArgs && ... args) {
515 auto method_type_info = &typeid(TResult (TOriginalClass::*)(TArgs ...));
516 auto i = m_map_name_behavior.find(method_type_info);
517 if (i == m_map_name_behavior.end())
518 throw std::runtime_error("mock calling error: unexpected call '" + std::string(method_type_info->name()) + "'");
519
520 try {
521 using TMethod = std::function<TResult (TArgs ...)>;
522 auto method = std::any_cast<TMethod>(i->second);
523 return method(std::forward<TArgs>(args)...);
524 } catch (const std::bad_any_cast&) {
525 throw std::runtime_error("mock calling error: type mismatch for method '" + std::string(method_type_info->name()) + "'");
526 }
