27//
28// ON_CALL(mock_object, Method(...))
29// .With(...) ?
30// .WillByDefault(...);

33// once.

35// EXPECT_CALL(mock_object, Method(...))

39// .WillOnce(...) *
40// .WillRepeatedly(...) ?
41// .RetiresOnSaturation() ? ;
42//

173 bool isEqual(const std::any &stored_any, TCallArgs &&... args) {
174 using TTuple = std::tuple<std::decay_t<TCallArgs>...>;
175
176 if (stored_any.type() != typeid(TTuple)) {
177 return false;
178 }
179
180 const TTuple &stored_tuple = std::any_cast<const TTuple&>(stored_any);
181 TTuple current_tuple(std::forward<TCallArgs>(args)...);
182
183 return stored_tuple == current_tuple;

265template<typename ... TArgs>
266::std::shared_ptr<TClass> Mock<TClass>::create(TArgs && ... args) {
267 if (m_is_mocked) {
268 if (m_mock_objects.empty()) {
269 throw std::runtime_error(
270 "getting regeistered mock of class '" + std::string(typeid(TClass).name()) +
271 "' object error: not enough such registered mock objects");
272 } else {
273 auto mock = m_mock_objects.front();
274 m_mock_objects.pop_front();
275 // shared_ptr without destructor
276 return std::shared_ptr<TClass>(dynamic_cast<TClass *>(mock), [] (TClass *) {});
277 }
278 } else {
279 return ::std::make_shared<TClass>(std::forward<TArgs>(args)...);
280 }

285template<typename TResult, typename ... TArgs>
286auto Mock<TClass>::call(TArgs && ... args) {
287 auto method_type_info = &typeid(TResult ( TOriginalClass::*)(TArgs ...));
288 auto i = m_map_name_behavior.find(method_type_info);
289 if (i == m_map_name_behavior.end())
290 throw std::runtime_error("mock calling error: unexpected call '" + std::string(method_type_info->name()) + "'");
291
292 try {
293 using TMethod = std::function<TResult (TArgs ...)>;
294 auto method = std::any_cast<TMethod>(i->second);
295 return method(std::forward<TArgs>(args)...);
296 } catch (const std::bad_any_cast&) {
297 throw std::runtime_error("mock calling error: type mismatch for method '" + std::string(method_type_info->name()) + "'");
298 }
299}
