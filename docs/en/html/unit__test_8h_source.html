135template<typename TValue>
136void UnitTest::fail(
137 TValue const &left,
138 TValue const &right,
139 std::string const &,
140 std::string const &condition_source,
141 std::string const &line)
142{
143 std::string message;
144
145 try {
146 message = line
147 + "\n'" + condition_source + "'\n"
148 + "L: " + convertion::convert<std::string, TValue, true>(left)
149 + "\nR: " + convertion::convert<std::string, TValue, true>(right);
150 } catch (std::exception const &) {}
151
152 if (message.empty()) {
153 message = line
154 + "\n'" + condition_source
155 +"'\ntype: " + typeid(TValue).name()
156 + "\nsize: " + convertion::convert<std::string>(sizeof(TValue));
157 }
158
159 throw Exception(message);
160}
