117template<typename TItem>
118CWorkerPoolPusher<TItem>::CWorkerPoolPusher(std::string const &name, THandlers const &handlers)
119:
120 WorkerPoolBase (name),
121 m_queue (CAsyncQueue<TItem>::create())
122{
123 auto i = 0;
124 for (auto const &handler: handlers)
125 m_threads.push_back(
126 CThread::create(name + "[" + convert<string>(i++) + "]",
127 CWorkerPusherRunnable<TItem>::create(handler, m_queue)));
