95template<typename TState, typename TEvent>
96void FSM<TState, TEvent>::addTransition(
97 TEvent const &event,
98 TState const &from,
99 TState const &to,
100 THandler const &handler)
101{
102 auto key = std::make_pair(from, event);
103
104 if (m_table.find(key) != m_table.end()) {
105 throw std::runtime_error(
106 "fsm adding transition error: duplicate transition for state '" +
107 convertion::convert<std::string>(from) + "' and event '" +
108 convertion::convert<std::string>(event) + "'");
109 }
110
111 m_table[key] = TTransition{from, event, to, handler};
112}

115template<typename TState, typename TEvent>
116TState FSM<TState, TEvent>::doAction(TEvent const &event) {
117 auto key = std::make_pair(m_state, event);
118 auto i = m_table.find(key);
119
120 if (i == m_table.end())
121 throw std::runtime_error(
122 "fsm action transit error: transition not found for event '" +
123 convertion::convert<std::string>(event) + "' on state '" +
124 convertion::convert<std::string>(m_state) + "'");
125
126 if (i->second.handler)
127 i->second.handler();
128
129 m_state = i->second.to;
130
131 return m_state;
132}

136size_t FSM<TState, TEvent>::TPairStateEventHash::operator()(std::pair<TState, TEvent> const &p) const {
137 size_t h1 = std::hash<TState> {} (p.first);
138 size_t h2 = std::hash<TEvent> {} (p.second);
139 return h1 ^ (h2 + 0x9e3779b9 + (h1 << 6) + (h1 >> 2));
140};
