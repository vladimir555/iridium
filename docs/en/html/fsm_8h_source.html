65void FSM<TState, TEvent>::addTransition(
66 TEvent const &event,
67 TState const &from,
68 TState const &to,
69 THandler const &handler)
70{
71 auto key = std::make_pair(from, event);
72
73 if (m_table.find(key) != m_table.end()) {
74 throw std::runtime_error(
75 "fsm adding transition error: duplicate transition for state '" +
76 convertion::convert<std::string>(from) + "' and event '" +
77 convertion::convert<std::string>(event) + "'");
78 }
79
80 m_table[key] = TTransition{from, event, to, handler};

84template<typename TState, typename TEvent>
85TState FSM<TState, TEvent>::doAction(TEvent const &event) {
86 auto key = std::make_pair(m_state, event);
87 auto i = m_table.find(key);
88
89 if (i == m_table.end())
90 throw std::runtime_error(
91 "fsm action transit error: transition not found for event '" +
92 convertion::convert<std::string>(event) + "' on state '" +
93 convertion::convert<std::string>(m_state) + "'");
94
95 if (i->second.handler)
96 i->second.handler();
97
98 m_state = i->second.to;
99
100 return m_state;
101}
