83template<typename TValue>
84FileValue<TValue>::FileValue(std::string const &file_name, TValue const &default_value)
85:
86 m_file_name (file_name)
87{
88 try {
89 std::ifstream f(m_file_name);
90 if (f.good() && f.peek() != std::ifstream::traits_type::eof()) { // Check if file is good and not empty
91 std::string value_str;
92 // Read the whole file content into value_str. A more robust way might be needed for multi-line values.
93 // This implementation reads only the first "word" or until whitespace if TValue is not string.
94 // If TValue is std::string, it might also only read the first word.
95 // For robust whole-file-to-string, one would typically use rdbuf.
96 // However, we'll assume 'f >> value_str;' is the intended way based on original code,
97 // which implies values are single-word strings or types convertible from such.
98 f >> value_str;
99 if (!value_str.empty()) { // If something was read
100 m_value = convertion::convert<TValue>(value_str);
101 } else { // File might be empty or only whitespace
102 m_value = default_value;
103 }
104 } else {
105 m_value = default_value;
106 }
107 } catch (std::exception const &e) {
108 // If conversion or other errors occur, it might be better to ensure m_value is default_value
109 // For now, rethrow wrapped, but consider if default_value should be assigned here too.
110 throw std::runtime_error("read file '" + m_file_name + "' error: " + e.what());
111 }
112}

116void FileValue<TValue>::operator = (TValue const &value) {
117 try {
118 m_value = value;
119
120 std::ofstream f(m_file_name, std::ios::trunc); // Open in truncation mode to overwrite
121 if (!f.good()) {
122 throw std::runtime_error("could not open file for writing");
123 }
124 f << convertion::convert<std::string>(value);
125 if (!f.good()) { // Check after write
126 throw std::runtime_error("write operation failed");
127 }
128 } catch (std::exception const &e) {
129 // Avoid re-wrapping if it's already our specific runtime_error
130 if (dynamic_cast<const std::runtime_error*>(&e) &&
131 std::string(e.what()).find("write file '" + m_file_name + "' error: ") != std::string::npos) {
132 throw;
133 }
134 throw std::runtime_error("write file '" + m_file_name + "' error: " + e.what());
135 }
136}
