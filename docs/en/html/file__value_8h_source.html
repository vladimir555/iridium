99template<typename TValue>
100FileValue<TValue>::FileValue(std::string const &file_name, TValue const &default_value)
101:
102 m_file_name (file_name)
103{
104 try {
105 std::ifstream f(m_file_name);
106 if (f.good() && f.peek() != std::ifstream::traits_type::eof()) { // Check if file is good and not empty
107 std::string value_str;
108 // Read the whole file content into value_str. A more robust way might be needed for multi-line values.
109 // This implementation reads only the first "word" or until whitespace if TValue is not string.
110 // If TValue is std::string, it might also only read the first word.
111 // For robust whole-file-to-string, one would typically use rdbuf.
112 // However, we'll assume 'f >> value_str;' is the intended way based on original code,
113 // which implies values are single-word strings or types convertible from such.
114 f >> value_str;
115 if (!value_str.empty()) { // If something was read
116 m_value = convertion::convert<TValue>(value_str);
117 } else { // File might be empty or only whitespace
118 m_value = default_value;
119 }
120 } else {
121 m_value = default_value;
122 }
123 } catch (std::exception const &e) {
124 // If conversion or other errors occur, it might be better to ensure m_value is default_value
125 // For now, rethrow wrapped, but consider if default_value should be assigned here too.
126 throw std::runtime_error("read file '" + m_file_name + "' error: " + e.what());
127 }
128}

132void FileValue<TValue>::operator = (TValue const &value) {
133 try {
134 m_value = value;
135
136 std::ofstream f(m_file_name, std::ios::trunc); // Open in truncation mode to overwrite
137 if (!f.good()) {
138 throw std::runtime_error("could not open file for writing");
139 }
140 f << convertion::convert<std::string>(value);
141 if (!f.good()) { // Check after write
142 throw std::runtime_error("write operation failed");
143 }
144 } catch (std::exception const &e) {
145 // Avoid re-wrapping if it's already our specific runtime_error
146 if (dynamic_cast<const std::runtime_error*>(&e) &&
147 std::string(e.what()).find("write file '" + m_file_name + "' error: ") != std::string::npos) {
148 throw;
149 }
150 throw std::runtime_error("write file '" + m_file_name + "' error: " + e.what());
151 }
152}
