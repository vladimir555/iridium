192template<typename TMutex, bool const is_tracable>
193bool Synchronized<TMutex, is_tracable>::Locker::wait() {
194 if (m_s->m_is_waitable)
195 m_s->m_cv.wait(m_l);
196
197 if (is_tracable)
198 printf("%s WM -> %s\n%s:%i\n",
199 threading::IThread::getNameStatic().c_str(),
200 m_s->m_is_waitable ? "OK" : "interrupted",
201 m_file, m_line);
202
203 return m_s->m_is_waitable; // ----->
204}

207template<typename TMutex, bool const is_tracable>
208bool Synchronized<TMutex, is_tracable>::Locker::wait(std::chrono::nanoseconds const &timeout) {
209 bool result =
210 m_s->m_is_waitable &&
211 m_s->m_cv.wait_for(m_l, timeout) == std::cv_status::no_timeout;
212 if (is_tracable)
213 printf("%s WM -> %s\n%s:%i\n",
214 threading::IThread::getNameStatic().c_str(),
215 result ? "OK" : m_s->m_is_waitable ? "timeout" : "interrupted",
216 m_file, m_line);
217 return result; // ----->
218}
