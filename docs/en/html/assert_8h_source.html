205template<typename T, typename TException>
206[[nodiscard]] std::decay_t<T> assertSize(T &&values, size_t const &size, TException const &exception) {
207 static_assert(TContainerHasSizeMethod<std::decay_t<T>>::value, "Type T must have a size() method");
208
209 auto values_size = values.size();
