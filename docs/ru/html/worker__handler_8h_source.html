127template<typename TItem>
128void CWorkerPusherRunnable<TItem>::run(std::atomic<bool> &is_running) {
129 while (is_running) {
130 auto items = m_queue->pop(is_running);
131 if (!items.empty())
132 m_handler->handle(items);
133 }
