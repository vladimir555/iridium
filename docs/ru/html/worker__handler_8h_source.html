101template<typename TItem>
102void CWorkerPusherRunnable<TItem>::run(std::atomic<bool> &is_running) {
103 while (is_running) {
104 auto items = m_queue->pop(is_running);
105 if (!items.empty())
106 m_handler->handle(items);
107 }
108}
