180template<typename TValue>
181typename INodeType<TValue>::TNodesSharedPtr CNodeType<TValue>::slice(std::string const &path) const {
182 static std::string const DEFAULT_PATH_DELIMITER("/");
183
184 // todo: depth overflow test
185 typename INodeType<TValue>::TNodesSharedPtr nodes;
186
187 if (path.empty())
188 return nodes; // ----->
189
190 if (path == "/")
191 return m_nodes; // ----->
192
193 std::string child_path = path.substr(1, path.find(DEFAULT_PATH_DELIMITER, 1) - 1);
194 std::string next_path;
195
196 if ((child_path.size() + DEFAULT_PATH_DELIMITER.size()) <= path.size())
197 next_path = path.substr(child_path.size() + DEFAULT_PATH_DELIMITER.size());
198
199 if (next_path.empty()) {
200 for (auto &node: m_nodes)
201 if (node->getName() == child_path)
202 nodes.push_back(node);
203 } else {
204 for (auto &node: m_nodes)
205 if (node->getName() == child_path)
206 nodes.splice(nodes.end(), node->slice(next_path));
207 }
208
209 return nodes; // ----->
210}
