293template<typename TValue>
294typename INodeType<TValue>::TNodes CNodeType<TValue>::slice(std::string const &path) {
295 static std::string const DEFAULT_PATH_DELIMITER("/");
296
297 // todo: depth overflow test
298 typename INodeType<TValue>::TNodes nodes;
299
300 if (path.empty())
301 return nodes; // ----->
302
303 if (path == "/")
304 return m_nodes; // ----->
305
306 std::string child_path = path.substr(1, path.find(DEFAULT_PATH_DELIMITER, 1) - 1);
307 std::string next_path;
308
309 if ((child_path.size() + DEFAULT_PATH_DELIMITER.size()) <= path.size())
310 next_path = path.substr(child_path.size() + DEFAULT_PATH_DELIMITER.size());
311
312 if (next_path.empty()) {
313 for (auto &node: m_nodes)
314 if (node->getName() == child_path)
315 nodes.push_back(node);
316 } else {
317 for (auto &node: m_nodes)
318 if (node->getName() == child_path)
319 nodes.splice(nodes.end(), node->slice(next_path));
320 }
321
322 return nodes; // ----->
323}
