57#define DEFINE_ENUM(TEnum, ...) \
58class TEnum { \
59public: \
60 DISABLE_C26827 \
61 enum TEnumInternal { \
62 __VA_ARGS__, \
63 UNKNOWN = INT_MIN \
64 }; \
65 ENABLE_C26827 \
66 TEnum(): m_value(TEnumInternal::UNKNOWN) {} \
67 TEnum(TEnum const &e) = default; \
68 TEnum(TEnumInternal const &e): m_value(e) {} \
69 TEnum(int const &value): m_value(static_cast<TEnumInternal>(value)) {} \
70 TEnum(std::string const &s): m_value(UNKNOWN) { \
71 for (auto const &i: map_enum_string) { \
72 if (iridium::lowerCase(i.second) == iridium::lowerCase(s)) { \
73 m_value = i.first; \
74 break; \
75 } \
76 } \
77 } \
78 ~TEnum() = default; \
79 operator std::string() const { \
80 try { \
81 return map_enum_string.at(m_value); \
82 } catch (...) { \
83 return "UNKNOWN"; \
84 } \
85 } \
86 operator TEnumInternal() const { \
87 return m_value; \
88 } \
89private: \
90 class Enums { \
91 public: \
92 Enums(TEnum::TEnumInternal const &e) { \
93 index = e; \
94 enums.push_back(TEnum(e)); \
95 } \
96 Enums(int const &e) { \
97 index = e; \
98 enums.push_back(TEnum(static_cast<TEnum::TEnumInternal>(e))); \
99 } \
100 Enums() { \
101 index++; \
102 enums.push_back(TEnum(static_cast<TEnum::TEnumInternal>(index))); \
103 } \
104 ~Enums() = default; \
105 static std::list<TEnum> enums; \
106 static int index; \
107 }; \
108 TEnumInternal m_value; \
109 static std::map<TEnumInternal, std::string> const map_enum_string; \
110 static std::map<TEnumInternal, std::string> const generateMap() { \
111 auto m = std::map<TEnumInternal, std::string>(); \
112 Enums UNUSED __VA_ARGS__; \
113 auto i = Enums::enums.begin(); \
114 for (auto const &arg: iridium::split(#__VA_ARGS__, ",")) { \
115 std::string name = iridium::trim(iridium::split(arg, "=").front()); \
116 if (m.find(*i) == m.end()) \
117 (m)[*i++] = name; \
118 else \
119 throw std::runtime_error(std::string(#TEnum) + " map key collision " + (m)[*i] + " and " + name); \
120 } \
121 return std::forward<std::map<TEnumInternal, std::string> const>(m); \
122 } \
123public: \
124 static std::list<TEnum> const &getEnums() { \
125 return Enums::enums; \
126 } \
127 static TEnum convert(std::string const &s) { \
128 TEnum e(s); \
129 if (e == UNKNOWN) { \
130 auto message = "convert enum string '" + s + "' to enum " + std::string(#TEnum) + " error: valid enum keys:"; \
131 for (auto const &i: getEnums()) \
132 message += std::string(" ") + static_cast<std::string>(TEnum(i)) + " = " + iridium::convertion::convert<std::string>(static_cast<int>(i)); \
133 throw std::runtime_error(message); \
134 } else \
135 return e; \
136 } \
137 static std::string convert(TEnum const &e) { \
138 return static_cast<std::string>(e); \
139 } \
140 static std::string convert(TEnum::TEnumInternal const &e) { \
141 return convert(TEnum(e)); \
142 } \
143 static std::string convert(std::list<TEnum> const &enums_) { \
144 std::string result; \
145 for (auto const &e: enums_) \
146 result += convert(e) + ", "; \
147 if (result.size() > 2) \
148 result.resize(result.size() - 2); \
149 return result; \
150 } \
151 std::string convertToFlagsString() const { \
152 std::string result; \
153 for (auto const &f: TEnum::getEnums()) \
154 if (m_value & f) \
155 result += convert(f) + " "; \
156 return result; \
157 } \
158};
