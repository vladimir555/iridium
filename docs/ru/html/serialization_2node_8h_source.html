370template<typename TValue>
371NodeView<TValue>::NodeView(NodeView<void> const &parent, std::string const &name, TValue const &default_value) {
372 m_path = parent.m_path + "/" + name;
373
374 if (parent.m_node)
375 m_node = parent.m_node->getChild(name);
376 else
377 m_node = nullptr;
378
379 if(!m_node) {
380 m_node = parent.m_node->addChild(name, convertion::convert<std::string>(default_value));
381 m_is_default = true;
382 }
383}
