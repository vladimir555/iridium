345NodeView<TValue>::NodeView(
346 INode::TSharedPtr const &node,
347 std::string const &name,
348 std::string const &path)
349{
350 m_path = path + "/" + name;
351
352 if (node && node->getName() == name)
353 m_node = node;
354 else
355 throw std::runtime_error("node '" + m_path + "' not found"); // ----->
356}

372template<typename TValue>
373NodeView<TValue>::NodeView(NodeView<void> const &parent, std::string const &name, TValue const &default_value) {
374 m_path = parent.m_path + "/" + name;
375
376 if (parent.m_node)
377 m_node = parent.m_node->getChild(name);
378 else
379 m_node = nullptr;
380
381 if(!m_node) {
382 m_node = parent.m_node->addChild(name, convertion::convert<std::string>(default_value));
383 m_is_default = true;
384 }
385}
