125template<typename TResult, typename ... TArgs>
126TResult LazyCacher<TResult, TArgs ... >::cacheFunc(TFunc func, TArgs ... args) {
127 TArgsTuple args_tuple(args ...);
128
129 typename TArgsTupleResultMap::const_iterator i = m.find(args_tuple);
130 if (i == m.end()) {
131// cout << "func: " << std::get<0>(args_tuple) << endl;
132 TResult r = func(args ...); // Corrected to TResult for clarity, auto also works.
133 m[args_tuple] = r;
134// return std::move(r); // -----> Potential optimization, but depends on TResult type.
135 return r; // -----> Returns a copy or moves if TResult is movable by compiler.
136 } else {
137// cout << "cache: " << std::get<0>(args_tuple) << endl;
138 return i->second; // -----> Returns a copy of the cached result.
139 }
140}

155template<typename TResult, typename ... TArgs>
156void LazyCacher<TResult, TArgs ... >::clean(TResult const &result) {
157 auto i = m.begin();
158 while (i != m.end()) {
159 if (i->second == result)
160 i = m.erase(i); // Correctly gets next valid iterator after erase
161 else
162 i++;
163 }
