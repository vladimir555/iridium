136template<typename TValue>
137CachedValue<TValue>::CachedValue() {
138 m_value = std::make_shared<TValue>();
139 // Attempt to find or insert the default value in the cache
140 // This part seems to be missing in the original logic for default constructor,
141 // adding it for consistency with the parameterized constructor's caching behavior.
142 auto it = m.find(*this);
143 if (it == m.end()) {
144 m[*this] = m_value;
145 } else {
146 m_value = it->second;
147 }
148}

151template<typename TValue>
152template<typename TParam>
153CachedValue<TValue>::CachedValue(TParam const &value) {
154 // todo: out of memory handling
155 // Initial construction to allow hashing and lookup
156 m_value = std::make_shared<TValue>(value);
157 auto it = m.find(*this); // Look up based on the content of the new value
158 if(it == m.end()) { // Not found in cache
159 // todo: out of memory handling
160 // m_value is already the correct new value, just insert it into the map
161 m[*this] = m_value;
162 } else { // Found in cache
163 m_value = it->second; // Use the existing shared_ptr
164 }
165}

185template<typename TValue>
186bool CachedValue<TValue>::operator == (CachedValue const &cached_value) const {
187 // If both point to the same shared_ptr instance, they are definitionally equal (and fast check)
188 if (m_value == cached_value.m_value) return true;
189 // Otherwise, if both pointers are valid, compare the pointed-to values.
190 if (m_value && cached_value.m_value) return *m_value == *cached_value.m_value;
191 // If one is null and the other isn't, or both are null but not the same instance (covered by first check), they are not equal.
192 return false;
193}

196template<typename TValue>
197bool CachedValue<TValue>::operator < (CachedValue const &cached_value) const {
198 // Comparison requires valid pointers.
199 if (m_value && cached_value.m_value) return *m_value < *cached_value.m_value;
200 // Define behavior for nulls: consider null less than non-null, two nulls are not less than each other.
201 if (!m_value && cached_value.m_value) return true; // this is null, other is not
202 return false; // otherwise (this is not null and other is, or both are null)
203}
