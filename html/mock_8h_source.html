73//
74// ON_CALL(mock_object, Method(...))

76// .WillByDefault(...);
77//

79// once.

81// EXPECT_CALL(mock_object, Method(...))

83// .Times(...) ?

85// .WillOnce(...) *
86// .WillRepeatedly(...) ?
87// .RetiresOnSaturation() ? ;
88//

559template<typename ... TArgs>
560::std::shared_ptr<TClass> Mock<TClass>::create(TArgs && ... args) {
561 if (m_is_mocked) {
562 if (m_mock_objects.empty()) {
563 throw std::runtime_error(
564 "getting regeistered mock of class '" + std::string(typeid(TClass).name()) +
565 "' object error: not enough such registered mock objects");
566 } else {
567 auto mock = m_mock_objects.front();
568 m_mock_objects.pop_front();
569 // shared_ptr without destructor
570 return std::shared_ptr<TClass>(dynamic_cast<TClass *>(mock), [] (TClass *) {});
571 }
572 } else {
573 return ::std::make_shared<TClass>(std::forward<TArgs>(args)...);
574 }

578template<typename TClass>
579template<typename TResult, typename ... TArgs>
580auto Mock<TClass>::call(TArgs && ... args) {
581 auto method_type_info = &typeid(TResult (TOriginalClass::*)(TArgs ...));
582 auto i = m_map_name_behavior.find(method_type_info);
583 if (i == m_map_name_behavior.end())
584 throw std::runtime_error("mock calling error: unexpected call '" + std::string(method_type_info->name()) + "'");
585
586 try {
587 using TMethod = std::function<TResult (TArgs ...)>;
588 auto method = std::any_cast<TMethod>(i->second);
589 return method(std::forward<TArgs>(args)...);
590 } catch (const std::bad_any_cast&) {
591 throw std::runtime_error("mock calling error: type mismatch for method '" + std::string(method_type_info->name()) + "'");
592 }
593}
