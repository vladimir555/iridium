121template<typename TEvent, typename TState>
122TState CFSM<TEvent, TState>::doAction(TEvent const &event) {
123 for (auto const &i: m_transitions) {
124 if (i.event == event && i.from == m_state) {
125 if (i.handler)
126 i.handler();
127 m_state = i.to;
128 m_history.push_back(i);
129 if (m_history.size() > m_history_size)
130 m_history.pop_front();
131 return m_state; // ----->
132 }
133 }
134 throw std::runtime_error("fsm error: transition not found for state '" +
135 convertion::convert<std::string>(m_state) + "' with event '" +
136 convertion::convert<std::string>(event)); // ----->
137}
