48void CObservable<TEvent>::unsubscribe(typename IObserver<TEvent>::TWeakPtr const &observer) {
49 LOCK_SCOPE();
50 auto target = observer.lock();
51
52 m_observers.erase(
53 std::remove_if(
54 m_observers.begin(),
55 m_observers.end(),
56 [&target](const auto& weakObs) {
57 auto obs = weakObs.lock();
58 return obs && obs == target;
59 }
60 ),
61 m_observers.end()
62 );
63}
