159template<typename TMutex, bool const is_tracable>
160void Synchronized<TMutex, is_tracable>::interrupt() {
161 m_is_waitable = false;
162 m_cv.notify_all();
163}

184template<typename TMutex, bool const is_tracable>
185Synchronized<TMutex, is_tracable>::Locker::~Locker() {
186 if (is_tracable)
187 printf("%s UM\n%s:%i\n",
188 threading::IThread::getNameStatic().c_str(),
189 m_file, m_line);
190
191 m_l.unlock();
192 m_s->m_cv.notify_one();
193}

196template<typename TMutex, bool const is_tracable>
197bool Synchronized<TMutex, is_tracable>::Locker::wait() {
198 if (m_s->m_is_waitable)
199 m_s->m_cv.wait(m_l);
200
201 if (is_tracable)
202 printf("%s WM -> %s\n%s:%i\n",
203 threading::IThread::getNameStatic().c_str(),
204 m_s->m_is_waitable ? "OK" : "interrupted",
205 m_file, m_line);
206
207 return m_s->m_is_waitable; // ----->
208}

211template<typename TMutex, bool const is_tracable>
212bool Synchronized<TMutex, is_tracable>::Locker::wait(std::chrono::nanoseconds const &timeout) {
213 bool result =
214 m_s->m_is_waitable &&
215 m_s->m_cv.wait_for(m_l, timeout) == std::cv_status::no_timeout;
216 if (is_tracable)
217 printf("%s WM -> %s\n%s:%i\n",
218 threading::IThread::getNameStatic().c_str(),
219 result ? "OK" : m_s->m_is_waitable ? "timeout" : "interrupted",
220 m_file, m_line);
221 return result; // ----->
222}
