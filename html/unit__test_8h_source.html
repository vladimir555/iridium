105template<typename TValue>
106void UnitTest::fail(
107 TValue const &left,
108 TValue const &right,
109 std::string const &,
110 std::string const &condition_source,
111 std::string const &line)
112{
113 std::string message;
114
115 try {
116 message = line
117 + "\n'" + condition_source + "'\n"
118 + "L: " + convertion::convert<std::string>(left)
119 + "\nR: " + convertion::convert<std::string>(right);
120 } catch (std::exception const &) {}
121
122 if (message.empty()) {
123 message = line
124 + "\n'" + condition_source
125 +"'\ntype: '" + typeid(TValue).name()
126 +"'\nsize: " + convertion::convert<std::string>(sizeof(TValue));
127 }
128
129 throw Exception(message);
130}

186{
187 bool is_fail = false;
188

192 fail(condition_source, line);

194 } catch (...) {
